\section{Indices calculations}
\begin{itemize}
    \item $a^m \times a^n = a^{m+n}$
    \item $a^m \div a^n = a^{m-n}$
    \item $(a^m)^n = a^{mn}$
    \item $a^{\frac{m}{n}} = \sqrt[n]{a^m}$
\end{itemize}
