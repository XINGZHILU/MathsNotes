\section{SUVAT equations}
\begin{itemize}
    \item $s=ut+\frac{1}{2}at^2$
    \item $s=vt-\frac{1}{2}at^2$
    \item $v=u+at$
    \item $v^2=u^2+2as$
    \item $s=\frac{1}{2}(u+v)t$
\end{itemize}
