\subsection{Vieta's Law}
\begin{itemize}
	\item $x_1+x_2+\dots+x_n=-\dfrac{a_{n-1}}{a_n}$
	\item $(r_1r_2+r_1r_3+\dots+r_1r_n)+(r_2r_3+r_2r_3+\dots+r_2r_n)+\dots+r_{n-1}r_n=\dfrac{a_{n-2}}{a_n}$
	\item $r_1r_2\dots r_n=(-1)^n\dfrac{a_0}{a_n}$
\end{itemize}
