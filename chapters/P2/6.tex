\section{Identities}
\begin{itemize}
    \item $\tan \theta = \dfrac{\sin \theta}{\cos \theta}$
    \item $\sin^2 \theta + \cos^2 \theta = 1$
    \item $\tan^2 \theta + 1 = \sec^2 \theta$
    \item $\cot^2 \theta + 1 = \cosec^2 \theta$
\end{itemize}
